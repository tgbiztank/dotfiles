<h3 align="center">Install</h3>
<h6 align="center">Arch Linux</h6>
<pre align="center"><code>
    cd ~
    git clone https://github.com/tgbiztank/dotfiles.git
    git branch neovim
    cd script
    chmod +x install.sh
    ./install.sh
  </code></pre>
</br>
<h3 align="center">Uninstall</h3>
<pre align="center"><code>
    rm -rf ~/.config/nvim
    rm -rf ~/.local/share/nvim
    rm -rf ~/.cache/nvim
  </code></pre>
